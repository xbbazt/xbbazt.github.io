学生成绩管理系统（python）

· 2025-12-16 · 
students=[]

while True:
    print("=== 学生成绩管理系统 ===")
    print("1. 新增学生成绩")
    print("2. 查看所有学生")
    print("3. 计算平均分")
    print("4. 查看最高分/最低分")
    print("5. 退出")
    choice=input(f'请输入功能:')
    
    if choice=='1':
        #新增学生成绩
        name=input(f'请输入学生姓名：')
        score =input(f'请输入学生成绩：')

        #转换为为整数
        score = int(score)

        #建立一个数据结构
        student = {'name': name, 'score': score}
        students.append(student)

        print(f'新增成功')
    elif choice=='2':
        if len(students) == 0:
            print("暂无学生成绩")
        else:
            print("所有学生成绩如下：")
            for student in students:
                print(f"姓名: {student['name']}  成绩: {student['score']}")
    elif choice=='3':
        if len(students) == 0:
            print(f'没有学生成绩')
            continue
        total = 0
        for student in students:
            total += student['score']
        avg = total / len(students)
        print(f'平均分为：',avg)  
    elif choice=='4':
        if len(students) == 0:
            print('暂无学生成绩')
            continue

        max_score = students[0]['score']
        min_score = students[0]['score']

        for student in students:
            score = student['score']
            if score > max_score:
                max_score = score
            if score < min_score:
                min_score = score
        print(f"最高分是：{max_score}")
        print(f"最低分是：{min_score}")
    elif choice=='5':
        print(f'推出系统')
        break
    else:
        print(f'输入无效重新输入')
